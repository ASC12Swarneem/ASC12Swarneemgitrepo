import {Component} from '@angular/core';

@Component(
  {
    selector: 'app-root',
    template: `
    <table>
      <tr>
        <th>Name</th>
        <th>Age</th>
        <th>Gender</th>
        <th>City</th>
      </tr>
      <tr>
        <td>{{user.name}}</td>
        <td>{{user.age}}</td>
        <td>{{user.gender}}</td>
        <td>{{user.city}}</td>
      </tr>
      <tr>
        <td>{{user2.name}}</td>
        <td>{{user2.age}}</td>
        <td>{{user2.gender}}</td>
        <td>{{user2.city}}</td>
      </tr>
    </table>`,
    styles: `
    table,tr,td,th{
    border : 1px solid black;
    }
    `,
    standalone: true,
  }
)

user = {
  name:  "Swarneem",
  age: 23,
  gender: "Male",
  city: "Nagpur"
}

  user2 = {
  name:  "Rohit",
  age: 22,
  gender: "Male",
  city: "Nagpur"
}
}
